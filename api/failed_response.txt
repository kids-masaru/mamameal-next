# Pattern for items and quantities, including optional metadata after the quantity
item_quantity_pattern = re.compile(r"^(.*?)\s+(\d+(?:\+\d+)?)(?:\s+(?:12/\d{2}|未就組|年少|年中|年長|縦クラス|Co|Al|ГРЕЧЕ|LEEEE))?\s*$")

# Pattern for lines that are *only* a quantity (and optional metadata)
qty_only_pattern = re.compile(r"^\d+(?:\+\d+)?(?:\s+(?:12/\d{2}|未就組|年少|年中|年長|縦クラス|Co|Al|ГРЕЧЕ|LEEEE))?\s*$")

# Pattern for dates and age groups/classes (metadata lines)
metadata_pattern = re.compile(r"^(12/\d{2}|未就組|年少|年中|年長|縦クラス|満三歳児)$")

# List of strings that are considered client context/metadata, not part of item names
client_context_words = [
    "ごはん用保温箱", "ご飯\d+g", "用保温箱", "具材用保温箱", "スープ用保温箱", "うどんスープ",
    "スープ", "おわん", "皿", "ちゃわん", "ルウ", "ミニフィッシュ", "ごはん",
    "主菜", "副菜"
]

# List of item names to explicitly exclude from the final bento_items list
item_names_to_exclude = {
    "未就組", "年少", "年中", "年長", "縦クラス", "満三歳児",
    "Co", "Al", "ГРЕЧЕ", "LEEEE"
}

# Standardize client names based on keywords in the line
def extract_client_name_standardized(line):
    if "雙葉小学校" in line: return "雙葉小学校"

# Clean item name by stripping known suffixes/metadata.
def clean_item_name(item_name):
    item_name = item_name.strip()
    # Remove dates (like 12/10) if accidentally included
    item_name = re.sub(r"12/\d{2}", "", item_name).strip()
    
    # Remove age groups/classes if accidentally included
    for meta in ["未就組", "年少", "年中", "年長", "縦クラス", "満三歳児"]:
        if item_name.endswith(meta):
            item_name = item_name[:-len(meta)].strip()
            
    # Remove generic context words only if they form the entire item name
    if item_name in client_context_words or re.match(r"ご飯\d+g", item_name):
        return "" # Completely filter out if it's just a context word
    
    return item_name

# Specific parser for 雙葉小学校 due to its unique OCR flattening (items then quantities)
def parse_futaba_block(block_lines):
    items_in_ocr = []
    quantities_in_ocr = []
    
    for line in block_lines:
        line = line.strip()
        if not line or metadata_pattern.match(line): continue
        if any(re.search(word, line) for word in client_context_words): continue # Skip context lines
        
        if re.match(r"^\d+$", line): # Line contains only a number -> quantity
            quantities_in_ocr.append(line)
        else: # Likely an item name
            cleaned_item = clean_item_name(line)
            if cleaned_item and cleaned_item not in item_names_to_exclude:
                items_in_ocr.append(cleaned_item)
                
    results = []
    qty_idx = 0
    # Match items with quantities in sequence
    for item_name in items_in_ocr:
        if qty_idx < len(quantities_in_ocr):
            results.append((item_name, quantities_in_ocr[qty_idx]))
            qty_idx += 1
        elif item_name == "1": # Specific rule for item "1" often implying 1 quantity
            results.append((item_name, "1"))
        # Else, item without quantity is implicitly 0 or should not be counted.
        
    return results

final_clients_data = {}
all_bento_items_set = set()
current_client = None
last_potential_item_name = None
is_futaba_school_block = False
futaba_block_lines_buffer = []

lines = [line.strip() for line in ocr_text.split('\n') if line.strip()]

for i, line_raw in enumerate(lines):
    line = line_raw.strip()
    if not line:
        continue

    # 1. Identify client lines
    standardized_client_name = extract_client_name_standardized(line)

    if standardized_client_name:
        # If we were in a 雙葉小学校 block, process it before switching client context
        if is_futaba_school_block and futaba_block_lines_buffer:
            parsed_futaba = parse_futaba_block(futaba_block_lines_buffer)
            for item_name, quantity_str in parsed_futaba:
                student_count = 0
                teacher_count = 0
                if "+" in quantity_str:
                    parts = quantity_str.split('+')
                    student_count = int(parts[0])
                    teacher_count = int(parts[1])
                else:
                    student_count = int(quantity_str)
                
                if item_name: all_bento_items_set.add(item_name)
                if student_count > 0: current_client["student_meals"].append(student_count)
                if teacher_count > 0: current_client["teacher_meals"].append(teacher_count)
            futaba_block_lines_buffer = [] # Reset buffer

        # Set new client
        if standardized_client_name not in final_clients_data:
            final_clients_data[standardized_client_name] = {
                "name": standardized_client_name,
                "student_meals": [],
                "teacher_meals": []
            }
        current_client = final_clients_data[standardized_client_name]
        last_potential_item_name = None # Reset item candidate

        # Check if new client is 雙葉小学校
        is_futaba_school_block = (standardized_client_name == "雙葉小学校")
        continue # Client line processed

    # If currently in a 雙葉小学校 block, buffer lines for special processing
    if is_futaba_school_block:
        futaba_block_lines_buffer.append(line)
        continue # Process later as a block

    # For other clients:
    if not current_client:
        continue # Skip if no client context set yet

    # 2. Check for item name and quantity on the same line
    match_inline = item_quantity_pattern.match(line)
    if match_inline:
        item_name_str = match_inline.group(1).strip()
        quantity_str = match_inline.group(2)
        last_potential_item_name = None # Item processed

        student_count = 0
        teacher_count = 0
        parsed_item_name = ""

        # Specific handling for "ごはん 2K×1" pattern
        if item_name_str == "ごはん 2K" and quantity_str == "1":
            parsed_item_name = "ごはん 2K"
            student_count = 1
            teacher_count = 0
        # Specific handling for "予備+職員 3+5"
        elif item_name_str == "予備+職員" and "+" in quantity_str:
            parts = quantity_str.split('+')
            student_count = int(parts[0])
            teacher_count = int(parts[1])
            parsed_item_name = "予備" # "予備" as student item
            all_bento_items_set.add("職員") # "職員" as a separate bento item
        elif "+" in quantity_str:
            parts = quantity_str.split('+')
            student_count = int(parts[0])
            teacher_count = int(parts[1])
        else:
            student_count = int(quantity_str)
        
        if not parsed_item_name: # If not handled by special cases, clean and assign
            parsed_item_name = clean_item_name(item_name_str)

        # Adjust for '職員室' or '職員' being the item name (all go to teacher meals)
        if parsed_item_name in ["職員室", "職員"]:
            teacher_count += student_count
            student_count = 0
        
        if parsed_item_name and parsed_item_name not in item_names_to_exclude:
            all_bento_items_set.add(parsed_item_name)
        
        if student_count > 0:
            current_client["student_meals"].append(student_count)
        if teacher_count > 0:
            current_client["teacher_meals"].append(teacher_count)

    # 3. Check if the line is a standalone quantity for a previously identified item name
    elif last_potential_item_name and qty_only_pattern.match(line):
        quantity_str = line.split()[0] # Take only the quantity part before any metadata
        
        student_count = 0
        teacher_count = 0
        if "+" in quantity_str:
            parts = quantity_str.split('+')
            student_count = int(parts[0])
            teacher_count = int(parts[1])
        else:
            student_count = int(quantity_str)

        if last_potential_item_name in ["職員室", "職員"]:
            teacher_count += student_count
            student_count = 0

        if last_potential_item_name and last_potential_item_name not in item_names_to_exclude:
            all_bento_items_set.add(last_potential_item_name)
        
        if student_count > 0:
            current_client["student_meals"].append(student_count)
        if teacher_count > 0:
            current_client["teacher_meals"].append(teacher_count)
        
        last_potential_item_name = None # Item processed

    # 4. Check if the line is a potential item name (not metadata or client context)
    elif not metadata_pattern.match(line) and not any(re.search(word, line) for word in client_context_words):
        potential_item = clean_item_name(line)
        if potential_item and potential_item not in item_names_to_exclude:
            last_potential_item_name = potential_item
        else:
            last_potential_item_name = None

# Process any remaining 雙葉小学校 block after the loop ends
if is_futaba_school_block and futaba_block_lines_buffer:
    parsed_futaba = parse_futaba_block(futaba_block_lines_buffer)
